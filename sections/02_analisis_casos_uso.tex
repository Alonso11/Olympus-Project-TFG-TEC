\section{Análisis de Casos de Uso (Objetivo Específico 1)}

\subsection{Escenarios Operacionales del Sistema}
Para asegurar la cobertura funcional del Rover Olympus, se definen los escenarios operativos divididos en tres categorías críticas: Misión, Contingencia y Soporte.

\subsubsection{Operaciones de Misión (Flujo Nominal)}
\begin{enumerate}[label=\textbf{UC-NOM-\arabic*}]
    \item \textbf{Inicialización y Auto-diagnóstico (PBIT):} Verificación del estado de salud de sensores y actuadores al encendido.
    \item \textbf{Navegación Autónoma a Waypoint:} Cálculo y ejecución de trayectorias evitando obstáculos detectados por LiDAR/Cámaras.
    \item \textbf{Recolección de Datos Científicos:} Activación de carga útil y almacenamiento de datos en puntos de interés.
    \item \textbf{Transmisión de Telemetría:} Envío de estados del sistema y datos científicos a la estación base.
\end{enumerate}

\subsubsection{Gestión de Contingencias (Flujo de Fallos)}
\begin{enumerate}[label=\textbf{UC-CONT-\arabic*}]
    \item \textbf{Recuperación por Pérdida de Enlace:} Ejecución de protocolo de seguridad ante la interrupción de comunicaciones.
    \item \textbf{Respuesta a Baja Energía:} Transición a modo de bajo consumo y priorización de recarga.
\end{enumerate}

\subsubsection{Mantenimiento y Soporte}
\begin{enumerate}[label=\textbf{UC-SUP-\arabic*}]
    \item \textbf{Actualización de Software (OTA):} Recepción e instalación de parches críticos en el sistema de archivos del rover.
\end{enumerate}

\subsection{Descripción Detallada de Casos de Uso Críticos}

\begin{table}[H]
    \centering
    \begin{tabularx}{\textwidth}{|l|X|}
        \hline
        \textbf{ID} & UC-NOM-01 \\ \hline
        \textbf{Nombre} & Inicialización y Auto-diagnóstico (PBIT - Power-On Built-In Test) \\ \hline
        \textbf{Descripción} & El rover realiza una verificación sistemática de todos los subsistemas críticos al recibir energía. \\ \hline
        \textbf{Actor} & Sistema Interno / Operador (en caso de rearme manual). \\ \hline
        \textbf{Pre-condiciones} & Batería $>$ 20\%, Suministro de energía estable. \\ \hline
        \textbf{Flujo Principal} & 1. Inicio de microcontroladores de bajo nivel; 2. Verificación de integridad de sensores (IMU, LiDAR); 3. Comprobación de conectividad con motores; 4. Reporte de estado nominal a GCS. \\ \hline
        \textbf{Excepciones} & Si un sensor crítico falla, el rover entra en modo de "Fallo Crítico" y notifica a la base. \\ \hline
        \textbf{Condiciones Finales} & Rover listo para recibir comandos de navegación. \\ \hline
    \end{tabularx}
\end{table}
