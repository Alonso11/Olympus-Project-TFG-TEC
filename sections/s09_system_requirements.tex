\section{System Requirements}\label{sec:requirements}

Esta sección contiene la ``definición maestra'' de los requisitos del sistema (RF/RNF/CON) del Rover Olympus y sus derivados a nivel de subsistema, organizada según la arquitectura de 7 subsistemas. Los requisitos se consolidan aquí para asegurar la coherencia del baseline técnico. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}

\subsection{Functional Requirements (By Subsystem)}

Los requisitos funcionales describen capacidades obligatorias del Rover Olympus para ejecutar UC‑01. Se organizan bajo los 7 subsistemas técnicos definidos en la arquitectura.

\subsubsection{Structural (Estructura / Chasis)}
\textbf{Soporte de Sistema:}
\begin{itemize}
    \item \textbf{STR-REQ-001:} El subsistema Structural DEBERÁ soportar la masa total del rover manteniendo un factor de seguridad de al menos 1.5 frente a cargas estáticas en pendientes de hasta 20°. [EXTERNO]
    \item \textbf{STR-REQ-002:} El subsistema Structural DEBERÁ proveer una envolvente de protección para la electrónica crítica equivalente a IP4X para prevenir el ingreso de suelo volcánico. [EXTERNO]
\end{itemize}

\subsubsection{Tracción (Locomotion)}
\textbf{Soporte de RF-003, RF-004:}
\begin{itemize}
    \item \textbf{TRAC-REQ-001:} El subsistema de Tracción DEBERÁ permitir el desplazamiento estable en superficies volcánicas con pendientes de hasta 20°.
    \item \textbf{TRAC-REQ-002:} El subsistema de Tracción DEBERÁ proveer un despeje del suelo (ground clearance) de al menos 10 cm. [EXTERNO]
\end{itemize}

\subsubsection{Propulsión (Drive / Actuación)}
\textbf{Soporte de RF-005:}
\begin{itemize}
    \item \textbf{PROP-REQ-001:} El subsistema de Propulsión DEBERÁ generar torque suficiente para iniciar el movimiento desde el reposo en una pendiente de 20°.
    \item \textbf{PROP-REQ-002:} El subsistema de Propulsión DEBERÁ detener el giro de las ruedas en un tiempo $\le$ 500 ms tras una señal de Fault Stop (soporte a RF-005).
\end{itemize}

\subsubsection{C\&DH (Command \& Data Handling)}
\textbf{Soporte de RNF-001, RF-006:}
\begin{itemize}
    % CDH-REQ-001 (abstraído de Yocto)
    \item \textbf{CDH-REQ-001:} El subsistema C\&DH \textbf{DEBERÁ} garantizar un tiempo de respuesta del sistema ante comandos de navegación de \(\le 2\,\mathrm{s}\) (soporte a RNF-001).
    \item \textbf{CDH-REQ-002:} El subsistema C\&DH DEBERÁ disponer de almacenamiento local suficiente para 2 horas de telemetría y logs de misión. [EXTERNO]
\end{itemize}

\subsubsection{Energía (EPS / Power)}
\textbf{Soporte de CON-001:}
\begin{itemize}
    \item \textbf{EPS-REQ-001:} El subsistema de Energía DEBERÁ reportar voltaje y corriente al C\&DH con una frecuencia de $\ge$ 1 Hz y precisión de $\pm$ 1\% mediante sensores de monitoreo de potencia integrados.
    \item \textbf{EPS-REQ-002:} El subsistema de Energía DEBERÁ interrumpir la potencia a los motores ante voltajes críticos de la batería principal del sistema. [EXTERNO]
\end{itemize}

\subsubsection{COMM (Telecomunicaciones)}
\textbf{Soporte de RF-006, CON-002:}
\begin{itemize}
    \item \textbf{COMM-REQ-001:} El subsistema COMM DEBERÁ implementar el protocolo CSP para asegurar la integridad de TM/CMD (soporte a RF-006).
    \item \textbf{COMM-REQ-002:} El subsistema COMM DEBERÁ mantener un enlace estable a 2.4 GHz hasta 50 m de distancia en línea de vista. [EXTERNO]
\end{itemize}

% --- Requisitos UHF elevados a Nivel Sistema ---
\paragraph{COMM-REQ-003 (Enlace UHF Robusto).}
El subsistema COMM \textbf{DEBERÁ} proporcionar un enlace de telemetría y comando en banda UHF que permita la operación en condiciones de baja visibilidad (NLoS) mediante protocolos de retransmisión y entrega garantizada. \\
\textit{Verificación:} Test/Demo con casos directo y vía retransmisor.

\paragraph{COMM-REQ-004 (Integridad en UHF).}
El sistema \textbf{DEBERÁ} asegurar la integridad de los datos de telemetría y comando mediante mecanismos de validación de errores en múltiples capas durante la operación en UHF. \\
\textit{Verificación:} Capturas de datos mostrando validación de integridad.

\paragraph{COMM-REQ-005 (Conmutación de ruta).}
Ante la pérdida de enlace directo, el sistema \textbf{DEBERÁ} realizar una conmutación automática hacia rutas de retransmisión disponibles para mantener la conectividad. \\
\textit{Verificación:} Logs de conmutación y retorno a ruta directa.

\subsubsection{GNC (Guía, Navegación y Control)}
\textbf{Soporte de RF-001, RF-002, RF-003, RF-004:}
\begin{itemize}
    \item \textbf{GNC-REQ-001:} El subsistema GNC DEBERÁ adquirir imágenes mediante una interfaz de cámara de alta velocidad con un FOV horizontal $\ge$ 120° (soporte a RF-001).
    \item \textbf{GNC-REQ-002:} El subsistema GNC DEBERÁ clasificar la transitabilidad con una confianza $\ge$ 95\% en sectores de 10x10 cm (soporte a RF-002).
    \item \textbf{GNC-REQ-003:} El subsistema GNC DEBERÁ estimar el estado cinemático y deslizamiento del rover con una frecuencia de actualización mínima de 10 Hz (soporte a RF-004). [EXTERNO]
\end{itemize}

\subsection{System Functional Requirements (Definición Maestra)}

\begin{reqblock}{RF-001}{Percepción Visual}
    \ReqShall{El sistema DEBERÁ adquirir percepción visual del entorno frontal para navegación autónoma. Ver GNC-REQ-001.}
    \item[\textbf{Prioridad:}] CRÍTICO
    \ReqTrace{ConOps UC-01; §7.3.1}
    \item[\textbf{Racional:}] Base fundamental para la detección de obstáculos y planificación de trayectorias.
    \ReqAccept{Demonstration (Visualización en GCS)}
\end{reqblock}

\begin{reqblock}{RF-002}{Clasificación de Transitabilidad}
    \ReqShall{El sistema DEBERÁ clasificar la transitabilidad del terreno para navegación autónoma. Ver GNC-REQ-002.}
    \item[\textbf{Prioridad:}] CRÍTICO
    \ReqTrace{ConOps UC-01; §7.3.2}
    \item[\textbf{Racional:}] Permite al agente ELANav distinguir zonas seguras de zonas de riesgo (alto slip o colisión).
    \ReqAccept{Analysis / Test (Confianza $\ge$ 95\%)}
\end{reqblock}

\begin{reqblock}{RF-003}{Evitación Reactiva de Obstáculos}
    \ReqShall{El sistema DEBERÁ detectar y evitar obstáculos físicos con una altura $\ge$ 20 cm durante la navegación autónoma. Ver GNC-REQ-002.}
    \item[\textbf{Prioridad:}] CRÍTICO
    \ReqTrace{ConOps UC-01; §7.3.3}
    \item[\textbf{Racional:}] Prevenir daños estructurales y estancamiento en terrenos con irregularidades significativas.
    \ReqAccept{Test / Demonstration (Pruebas de campo)}
\end{reqblock}

\begin{reqblock}{RF-004}{Compensación de Slip}
    \textit{[\textbf{SUPERSEDED} por RF-004-R1 — CR-002. Texto original preservado para trazabilidad histórica del baseline.]}
    \ReqShall{El sistema DEBERÁ estimar y compensar dinámicamente el deslizamiento (slip) en pendientes volcánicas. Ver GNC-REQ-003.}
    \item[\textbf{Prioridad:}] ALTO
    \ReqTrace{ConOps UC-01; §7.3.4}
    \item[\textbf{Racional:}] Garantizar la fidelidad de la odometría y evitar la deriva del rover en superficies no consolidadas.
    \ReqAccept{Analysis (Comparación vs Ground Truth)}
\end{reqblock}

\begin{reqblock}{RF-004-R1}{Compensación de Slip (Encoder-Based)}
    \ReqShall{El sistema DEBERÁ estimar y compensar dinámicamente el deslizamiento (slip) en pendientes volcánicas mediante odometría de encoders (6 ruedas), actualizando el estado cinemático a $\ge$ 10 Hz. Reemplaza RF-004. Ver CR-002.}
    \item[\textbf{Prioridad:}] ALTO
    \ReqTrace{ConOps UC-01; §7.3.4; CR-002}
    \item[\textbf{Racional:}] EKF descartado del scope actual del TFG. La compensación se implementa mediante lectura de encoders en LLC (Arduino Mega), validando slip por comparación de velocidad angular estimada vs. velocidad real medida. La métrica de aceptación se define contra ground truth físico en sitio análogo.
    \ReqAccept{Analysis/Test (Encoders vs. Ground Truth — distancia estimada vs. medida real)}
\end{reqblock}

\begin{reqblock}{RF-005}{Fault Stop Resiliente}
    \ReqShall{El sistema DEBERÁ ejecutar una transición inmediata a estado seguro (Stop) ante la detección de slip $>$ 25\% o fallos de navegación. Ver PROP-REQ-002.}
    \item[\textbf{Prioridad:}] CRÍTICO
    \ReqTrace{ConOps UC-01; §7.3.5}
    \item[\textbf{Racional:}] Proteger la integridad del rover ante condiciones de tracción inmanejables.
    \ReqAccept{Demonstration (Simulación de fallo)}
\end{reqblock}

\begin{reqblock}{RF-006}{Integridad de Telemetría}
    \ReqShall{El sistema DEBERÁ asegurar la integridad y consistencia de los datos de telemetría y comando (TM/CMD) durante todo el ciclo de misión. Ver COMM-REQ-001.}
    \item[\textbf{Prioridad:}] CRÍTICO
    \ReqTrace{ConOps UC-01; §7.3.6}
    \item[\textbf{Racional:}] Evitar la corrupción de comandos críticos de navegación y garantizar la veracidad de la telemetría recibida en la GCS.
    \ReqAccept{Analysis (Logs de CRC/CSP)}
\end{reqblock}

\subsection{Secondary Use Cases Requirements (UC-02 to UC-05)}
Conforme a la identificación de vacíos en la auditoría técnica, se establecen los requerimientos funcionales maestros para los casos de uso secundarios que garantizan la resiliencia y seguridad del sistema.

\subsubsection{UC-02: Homing and Retreat}
\begin{itemize}
    \item \textbf{SYS-FUN-020:} El sistema DEBERÁ mantener un registro dinámico de los últimos 5 waypoints transitables (puntos seguros) durante la navegación.
    \item \textbf{SYS-FUN-021:} El sistema DEBERÁ iniciar una maniobra de retorno autónomo (Retreat) ante una pérdida de enlace $>$ 10 s o un deslizamiento persistente $>$ 25\%.
\end{itemize}

\subsubsection{UC-03: Manual Override}
\begin{itemize}
    \item \textbf{SYS-FUN-030:} El sistema DEBERÁ permitir la interrupción inmediata de los modos autónomos mediante comandos de control manual prioritarios.
    \item \textbf{SYS-FUN-031:} El tiempo de transición entre modo autónomo y control manual NO DEBERÁ exceder los 500 ms tras la recepción del comando.
\end{itemize}

\subsubsection{UC-04: Safe Mode}
\begin{itemize}
    \item \textbf{SYS-FUN-040:} \textit{[\textbf{SUPERSEDED} por SYS-FUN-040a y SYS-FUN-040b. Texto original preservado para trazabilidad.]} El sistema DEBERÁ entrar en \emph{Safe Mode} automáticamente ante voltajes de batería críticos ($<$ 12V) o fallos de latencia en el C\&DH $>$ 5 s.
    \item \textbf{SYS-FUN-040a:} El sistema DEBERÁ entrar en \emph{Safe Mode} automáticamente ante voltaje de batería crítico ($<$ 12V), detectado por el monitor EPS (INA226).
    \item \textbf{SYS-FUN-040b:} El sistema DEBERÁ entrar en \emph{Safe Mode} automáticamente ante fallos de latencia en el C\&DH superiores a 5\,s sin respuesta del ciclo de agente HLC.
    \item \textbf{SYS-FUN-041:} En \emph{Safe Mode}, el sistema DEBERÁ priorizar la telemetría de salud y desenergizar los actuadores de tracción.
\end{itemize}

\subsubsection{UC-05: Safe Deactivation}
\begin{itemize}
    \item \textbf{SYS-FUN-050:} El sistema DEBERÁ sincronizar y cerrar todos los logs de misión en almacenamiento no volátil antes de confirmar la disponibilidad de apagado.
    \item \textbf{SYS-FUN-051:} El sistema DEBERÁ asegurar que los actuadores se encuentran en una configuración estable (parking) antes de la desactivación final.
\end{itemize}

\subsection{Performance and Non-Functional Requirements}
\sysreq{RNF-001}{Latencia de Procesamiento}{CRÍTICO}{ELANav}{$\le$ 2 s por ciclo.}
\sysreq{RNF-002}{Resiliencia de Recuperación}{ALTO}{ConOps}{Éxito $\ge$ 90\% en rescates.}
\sysreq{RNF-003}{Precisión de Navegación}{ALTO}{ConOps}{Error $<$ 5\% dist. total.}

\begin{reqblock}{RNF-004}{Límite Térmico Estructural}
    \ReqShall{El sistema DEBERÁ operar en un entorno cuya temperatura ambiente no exceda los 45 °C, garantizando que el chasis no supere los 60 °C.}
    \item[\textbf{Prioridad:}] ALTO
    \ReqTrace{Análisis Térmico §8.2}
    \item[\textbf{Racional:}] Prevenir la deformación estructural del material termoplástico (PLA MAX) en condiciones de calor extremo.
    \ReqAccept{Analysis / Test (Cámara térmica)}
\end{reqblock}

\begin{reqblock}{RNF-005}{Compatibilidad Electromagnética (EMC)}
    \ReqShall{El sistema DEBERÁ minimizar la interferencia electromagnética entre los actuadores de potencia y los sensores de navegación mediante blindaje y cableado trenzado.}
    \item[\textbf{Prioridad:}] MEDIO
    \ReqTrace{Mejores prácticas NASA SE}
    \item[\textbf{Racional:}] Evitar ruidos en la IMU y GPS causados por los picos de corriente de los motores DC.
    \ReqAccept{Inspection / Test}
\end{reqblock}

\begin{reqblock}{RNF-006}{Estabilidad de Regulación Energética}
    \ReqShall{El sistema DEBERÁ mantener las líneas de voltaje crítico (5V y 12V) con una estabilidad de $\pm$ 5\% bajo carga máxima.}
    \item[\textbf{Prioridad:}] CRÍTICO
    \ReqTrace{EPS-REQ-001}
    \item[\textbf{Racional:}] Garantizar que el computador de a bordo (OBC) no sufra reinicios por caídas de tensión (brown-outs) durante el arranque de motores.
    \ReqAccept{Analysis (Logs de EPS)}
\end{reqblock}

\begin{reqblock}{RNF-007}{Duración Operacional Mínima (MOD)}
    \ReqShall{El sistema DEBERÁ operar de forma continua sin fallo de misión durante al menos 2 horas en condiciones nominales de campaña (UC-01 activo, telemetría habilitada, navegación autónoma).}
    \item[\textbf{Prioridad:}] ALTO
    \ReqTrace{CDH-REQ-002; ConOps Fase 4}
    \item[\textbf{Racional:}] Define la duración mínima de operación ininterrumpida para una campaña UC-01, alineada con la capacidad de almacenamiento de telemetría (CDH-REQ-002: 2 horas). Un fallo de misión se define como cualquier transición no planificada a SafeMode o FAULT que detenga la navegación autónoma. \textit{Nota: esta métrica define una duración operacional mínima (MOD), no un MTBF estadístico; la estimación de MTBF queda diferida a análisis post-campaña con múltiples runs.}
    \ReqAccept{Demonstration (Ejecución continua UC-01 $\ge$ 2 h sin transición no planificada a SafeMode/FAULT)}
\end{reqblock}

\subsubsection{Environmental Requirements (ISO 29148 §9.5.9)}\label{subsubsec:env_req}

Esta subsección consolida los requisitos ambientales operacionales y de transporte del Rover Olympus conforme a ISO/IEC/IEEE 29148 §9.5.9 y ECSS-E-ST-10-06C §5.4. Los requisitos RNF-004 (temperatura máxima), RNF-005 (EMC cualitativo) y STR-REQ-002 (IP4X) se complementan con los ENV-REQ definidos aquí, que cierran las brechas de cota inferior, humedad, método de test y EMC cuantitativo. \cite{nasa_cm, segoldmine_ppi}

\begin{itemize}
    \item \textbf{ENV-REQ-001 (Temperatura operacional mínima):} El sistema DEBERÁ operar correctamente con una temperatura ambiente mínima de $+$5\,°C, conforme a los límites operacionales de los componentes COTS críticos (RPi5: 0–50\,°C; Arduino Mega: $-$40–85\,°C; celdas LiPo: 0–45\,°C en descarga). El límite operacional del sistema queda fijado en $+$5\,°C por la restricción de las celdas LiPo. \\
    \textit{Verificación:} Analysis (hojas de datos COTS; confirmar rango en datasheet de cada componente crítico).

    \item \textbf{ENV-REQ-002 (Humedad relativa):} El sistema DEBERÁ operar en un rango de humedad relativa de 20\,\% a 90\,\% sin condensación. Las PCBs expuestas DEBERÁN estar revestidas con barniz conformal o equivalente. \\
    \textit{Verificación:} Inspection (verificación visual de revestimiento conformal en PCBs; análisis de datasheet para componentes sensibles).

    \item \textbf{ENV-REQ-003 (Protección antipartículas — IP4X con método):} El subsistema STR DEBERÁ proveer protección equivalente a IP4X (IEC 60529) para la electrónica crítica. Método de test aceptado: exposición al flujo de aire con polvo volcánico simulado ($\le$1\,mm de diámetro) durante 30\,min a caudal moderado, seguida de inspección visual de ausencia de partículas en PCBs. Complementa STR-REQ-002. \\
    \textit{Verificación:} Test (blow test con polvo simulado + inspección post-exposición).

    \item \textbf{ENV-REQ-004 (Shock de transporte):} El sistema DEBERÁ sobrevivir sin pérdida de funcionalidad a una caída libre de 0.5\,m sobre superficie firme en su contenedor de transporte (PKG-REQ-001), simulando condiciones de traslado al sitio análogo. \\
    \textit{Verificación:} Test (drop test desde 0.5\,m; verificar boot completo y ausencia de daños estructurales post-caída).

    \item \textbf{ENV-REQ-005 (EMC cuantitativo):} La interferencia electromagnética generada por los actuadores de potencia NO DEBERÁ causar errores de lectura ($>$5\,\% fuera de rango nominal) en los sensores de navegación con interfaz I$^2$C o GPIO a distancias de cableado $\ge$10\,cm con blindaje aplicado. Complementa y cuantifica RNF-005. \\
    \textit{Verificación:} Test (lectura INA226 y HC-SR04 con motores a plena carga; verificar que las lecturas estén dentro del $\pm$5\,\% del valor de referencia sin motores).
\end{itemize}

\subsubsection{Physical Characteristics (ISO 29148 9.5.11)}
\begin{itemize}
    \item \textbf{PHY-REQ-001 (Dimensiones):} El rover DEBERÁ tener unas dimensiones nominales de envolvente de 480 x 340 x 320 mm ($\pm$ 10\%).
    \item \textbf{PHY-REQ-002 (Masa):} La masa total del sistema en configuración de vuelo (incluyendo baterías) NO DEBERÁ exceder los 7.5 kg.
    \item \textbf{PHY-REQ-003 (Centro de Masa):} El centro de masa DEBERÁ situarse por debajo de la línea de los ejes del rocker-bogie para maximizar la estabilidad en pendientes de 20°.
\end{itemize}

\subsection{Usability, Security, and Life Cycle Requirements}

\subsubsection{Usability (ISO 29148 9.5.6)}
\begin{itemize}
    \item \textbf{USA-REQ-001:} La interfaz de la Estación de Control Terreno (GCS) DEBERÁ ser legible bajo condiciones de luz solar directa (entorno análogo).
    \item \textbf{USA-REQ-002:} El sistema DEBERÁ permitir el despliegue operativo (desde transporte hasta inicio de misión) en un tiempo $\le$ 15 minutos.
    \item \textbf{USA-REQ-003:} El sistema DEBERÁ proporcionar alertas visuales/auditivas claras ante eventos críticos de \emph{Fault Stop}.
\end{itemize}

\subsubsection{System Security and Cybersecurity (ISO 29148 9.5.13)}
\begin{itemize}
    \item \textbf{SEC-REQ-001:} El sistema DEBERÁ procesar únicamente comandos provenientes de nodos de control autorizados y autenticados en la red del sistema.
    \item \textbf{SEC-REQ-002:} Los parámetros críticos de navegación DEBERÁN estar protegidos contra modificaciones no autorizadas durante la ejecución de la misión.
\end{itemize}

\subsubsection{Information Management (ISO 29148 9.5.14)}
\begin{itemize}
    \item \textbf{INF-REQ-001:} El sistema DEBERÁ almacenar logs de telemetría y eventos en memoria no volátil, asegurando la persistencia tras pérdida de energía.
    \item \textbf{INF-REQ-002:} Todos los registros de datos DEBERÁN estar sincronizados temporalmente con una precisión de $\le$ 100 ms respecto al reloj maestro.
\end{itemize}

\subsubsection{Life Cycle and Packaging (ISO 29148 9.5.16/17)}
\begin{itemize}
    \item \textbf{LCS-REQ-001:} Los componentes críticos (baterías y sensores) DEBERÁN ser accesibles y reemplazables en campo en $\le$ 20 minutos con herramientas estándar.
    \item \textbf{PKG-REQ-001:} El rover DEBERÁ contar con un contenedor rígido de transporte que lo proteja de vibraciones y humedad durante el traslado.
    \item \textbf{PKG-REQ-002:} El sistema DEBERÁ soportar temperaturas de almacenamiento de -10°C a +40°C sin degradación permanente de la electrónica o baterías.
\end{itemize}
