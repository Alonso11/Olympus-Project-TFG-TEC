\subsection{Verificación de requerimientos}

\subsubsection{Métodos de verificación}
\begin{itemize}
  \item \textbf{Inspección (I):} verificación por revisión visual/documental.
  \item \textbf{Análisis (A):} verificación por cálculo o análisis de datos.
  \item \textbf{Demostración (D):} verificación por ejecución observable en entorno representativo.
  \item \textbf{Prueba/Test (T):} verificación por ensayo instrumentado con mediciones cuantitativas.
\end{itemize}
