\section{Verification, Validation, and Acceptance}\label{sec:vv}

Esta sección define la estrategia de Verificación y Validación (V\&V) y los criterios de aceptación del Rover Olympus, asegurando que los requisitos RF/RNF/CON (Sección~\ref{sec:requirements}) se puedan demostrar con evidencia objetiva durante campañas en entorno análogo. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
La V\&V se estructura alrededor del Caso de Uso núcleo UC‑01 (Exploración de superficie) y sus contingencias (p.~ej., slip alto, pérdida de enlace). Se aplican métodos estándar T/A/D/I (Test/Analysis/Demonstration/Inspection) y se conservan las métricas cuantitativas definidas en los borradores ($\ge$95\% transitabilidad, 0 colisiones, slip $>$20\% $\rightarrow$ stop $<$5 s, latencia $\le$2 s, error $<$5\%, éxito rescate $\ge$90\%). \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
El objetivo de esta sección es:
\begin{itemize}
    \item \textbf{(i)} Establecer cómo se comprobará cada requisito.
    \item \textbf{(ii)} Definir el ``entorno representativo'' para validación.
    \item \textbf{(iii)} Declarar condiciones mínimas para declarar el sistema ``aceptado'' para la finalidad académica de validación de ELANav. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}
\end{itemize}

\subsection{V\&V Strategy Overview}

\textbf{Estrategia general:}
\begin{enumerate}
    \item \textbf{Verificación (Verification):} Demostrar que el sistema cumple cada requisito RF/RNF/CON mediante evidencias objetivas (métodos T/A/D/I). \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
    \item \textbf{Validación (Validation):} Demostrar que el sistema satisface su propósito operacional (validar ELANav en un rover de exploración) ejecutando UC‑01 en escenarios representativos del entorno análogo (pendientes hasta 20°, transiciones de suelo y latencia simulada 0–5 s) y alcanzando las métricas definidas. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
\end{enumerate}

\textbf{Principios de aplicación:}
\begin{itemize}
    \item \textbf{Enfoque por escenarios:} UC‑01 es el escenario dominante; las contingencias (slip alto, pérdida de enlace) se usan como casos de validación de robustez y resiliencia (RF‑005, RNF‑002). \cite{segoldmine_ppi, nasa_cm, omg_sysml_mbse}  
    \item \textbf{Evidencia cuantitativa:} Se priorizan criterios medibles ya establecidos (p. ej., $\ge$95\% transitabilidad, $\le$2 s latencia, $<$5\% error navegación). \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
    \item \textbf{Trazabilidad V\&V:} Cada requisito se conecta a (a) fuente ConOps/ELANav, (b) subsistema/arquitectura (BDD/PBS/ASM/N2) y (c) evidencia V\&V; esta trazabilidad se mantiene mediante la matriz de cumplimiento del proyecto. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}
\end{itemize}

\subsection{Verification Methods}

Se adoptan los métodos de verificación estándar indicados en los borradores:
\begin{itemize}
    \item \textbf{T: Test (Prueba)}
    \item \textbf{A: Analysis (Análisis)}
    \item \textbf{D: Demonstration (Demostración)}
    \item \textbf{I: Inspection (Inspección)} \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
\end{itemize}

\textbf{Asignación de métodos (resumen operativo):}
\begin{itemize}
    \item \textbf{RF‑001 (Percepción Visual CSI):} T/D para demostrar adquisición y cobertura (FOV $\ge$ 120°). \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \item \textbf{RF‑002 (Clasificación de Transitabilidad):} T/A para medir rendimiento ($\ge$ 95\% sectores). \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \item \textbf{RF‑003 (Evitación Reactiva):} D/T para demostrar cero colisiones con obstáculos $\ge$ 20 cm. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \item \textbf{RF‑004-R1 (Compensación de Slip — Encoder):} A/T para cuantificar error de odometría mediante encoders vs. ground truth (distancia real medida). Ver CR-002. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}
    \item \textbf{RF‑005 (Fault Stop):} T/D para confirmar activación ante slip $>$ 20\% en $<$ 5 s. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \item \textbf{RF‑006 (Integridad TM/CSP):} I/T para validar encapsulamiento CSP e integridad (p. ej., CRC‑32 válido). \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
    \item \textbf{RNF‑001 (Latencia):} T/A para medir $\le$ 2 s por ciclo de agente. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \item \textbf{RNF‑002 (Resiliencia):} D/T para demostrar éxito $\ge$ 90\% en rescates. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \item \textbf{RNF‑003 (Precisión navegación):} T/A para demostrar error $<$ 5\% dist. total. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \item \textbf{RNF‑007 (MTBF $\ge$ 2 h):} D para demostrar operación continua $\ge$ 2 h sin transición no planificada a SafeMode/FAULT durante ejecución de UC-01. \cite{nasa_cm, segoldmine_ppi}
    \item \textbf{CON‑001 / CON‑002:} I/D para confirmar condiciones de campaña (lab‑only; latencia 0–5 s aplicada). \cite{nasa_cm, segoldmine_ppi}
\end{itemize}

\subsection{Validation in Analog Environments}

La validación del Rover Olympus se realiza en un entorno análogo terrestre definido por el proyecto, diseñado para ejercer las capacidades clave de UC‑01 en condiciones representativas:

\textbf{Criterio de representatividad mínima del entorno (baseline):}
\begin{itemize}
    \item El entorno de validación DEBERÁ incluir al menos 3 transiciones de suelo (compacto $\rightarrow$ arena $\rightarrow$ pendiente) para forzar cambios de transitabilidad y condiciones de slip. \cite{omg_sysml_mbse, segoldmine_ppi}  
    \item El entorno DEBERÁ incluir pendientes de hasta 20° y obstáculos representativos para evaluar evitación reactiva y desempeño de navegación. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
    \item La campaña DEBERÁ permitir latencia simulada 0–5 s (CON‑002) cuando el objetivo de validación incluya emular condiciones de enlace operacional. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
\end{itemize}

\textbf{Enfoque de validación:}
\begin{itemize}
    \item Se valida el ``propósito'' del sistema demostrando UC‑01 completo cumpliendo simultáneamente criterios de percepción (RF‑001), transitabilidad (RF‑002), evitación (RF‑003), slip/FDIR (RF‑004/RF‑005), integridad TM (RF‑006) y desempeño (RNF‑001/RNF‑003), además de resiliencia (RNF‑002) bajo contingencias. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}
\end{itemize}

\subsection{Acceptance Criteria}

Se considera que el Rover Olympus es ``aceptado'' para el propósito de este SyRS (validación académica de ELANav) cuando:

\begin{enumerate}
    \item \textbf{Cumplimiento de requisitos núcleo (UC‑01):}
    \begin{itemize}
        \item \textbf{RF‑001:} Se demuestra percepción visual CSI con FOV $\ge$ 120°. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
        \item \textbf{RF‑002:} Se demuestra clasificación de transitabilidad correcta en $\ge$ 95\% de sectores. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
        \item \textbf{RF‑003:} Se demuestra evitación reactiva con 0 colisiones con obstáculos $\ge$ 20 cm. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
        \item \textbf{RF‑004-R1:} Se demuestra compensación de slip mediante encoders con error de odometría $<$ 5\% respecto a ground truth en recorrido de referencia. Ver CR-002. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}
        \item \textbf{RF‑005:} Se demuestra fault stop ante slip $>$ 20\% en $<$ 5 s. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
        \item \textbf{RF‑006:} Se demuestra integridad de telemetría (CSP) con paquetes válidos. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse}  
    \end{itemize}

    \item \textbf{Cumplimiento de desempeño y calidad:}
    \begin{itemize}
        \item \textbf{RNF‑001:} Latencia $\le$ 2 s por ciclo de agente. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
        \item \textbf{RNF‑003:} Precisión de navegación con error $<$ 5\% de la distancia total. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
        \item \textbf{RNF‑002:} Éxito $\ge$ 90\% en maniobras de rescate/recuperación ante contingencias definidas. \cite{omg_sysml_mbse, nasa_cm, segoldmine_ppi}  
    \end{itemize}

    \item \textbf{Condiciones de campaña:}
    \begin{itemize}
        \item Se verifica CON‑001 (marco lab‑only/sitio análogo) y se aplica CON‑002 (latencia simulada 0–5 s) cuando corresponda. \cite{nasa_cm, segoldmine_ppi}  
    \end{itemize}

    \item \textbf{Evidencia y trazabilidad:}
    \begin{itemize}
        \item Se entrega evidencia V\&V asociada a cada requisito, con método T/A/D/I documentado y trazabilidad a los escenarios de ConOps. \cite{nasa_cm, segoldmine_ppi, omg_sysml_mbse} 
    \end{itemize}
\end{enumerate}

\subsubsection*{V\&V--COMMS--01 (UHF/RDP directo)}
\textbf{Objetivo.} Demostrar TM/CMD fiable por UHF usando CSP/RDP punto--punto (Rover$\leftrightarrow$GCS).
\textbf{Criterio.} Entrega $\ge 95\%$ de N tramas (p.\,ej.\ 32/64/128/256\,B) y decodificación CSP + CRC--16 (AX.25).
\textbf{Evidencia.} Capturas/pcap y consola KISS; trazas RDP (ACK/reintentos).

\subsubsection*{V\&V--COMMS--02 (UHF vía retransmisor, store--and--forward)}
\textbf{Objetivo.} Validar el flujo Rover$\rightarrow$Relay$\rightarrow$GCS con rutas CSP.
\textbf{Criterio.} Entrega $\ge 95\%$ de N tramas en RDP; confirmación de reenvío en el nodo intermedio.
\textbf{Evidencia.} Logs del relay (recepción/reenvío/ACK), trazas CSP y rutas.
